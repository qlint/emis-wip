CREATE TABLE app.schoolbus_trips
(
  schoolbus_trip_id serial NOT NULL,
  trip_name character varying,
  trip_routes character varying,
  bus_id integer,
  creation_date timestamp without time zone NOT NULL DEFAULT now(),
  modified_date timestamp without time zone DEFAULT now(),
  CONSTRAINT "PK_schoolbus_trip_id" PRIMARY KEY (schoolbus_trip_id ),
  CONSTRAINT bus_id FOREIGN KEY (bus_id)
      REFERENCES app.buses (bus_id) MATCH FULL
      ON UPDATE NO ACTION ON DELETE NO ACTION
)
WITH (
  OIDS=FALSE
);
ALTER TABLE app.schoolbus_trips
  OWNER TO postgres;
