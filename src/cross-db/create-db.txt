CREATE TABLE app.school_resources
(
    resource_id serial NOT NULL,
    class_id integer,
    term_id integer,
    emp_id integer,
    resource_name character varying COLLATE pg_catalog."default",
    resource_type character varying COLLATE pg_catalog."default",
    file_name character varying COLLATE pg_catalog."default",
    additional_text character varying COLLATE pg_catalog."default",
    active boolean NOT NULL DEFAULT true,
    creation_date timestamp without time zone NOT NULL DEFAULT now(),
    modified_date timestamp without time zone,
    CONSTRAINT "PK_resource_id" PRIMARY KEY (resource_id)
)

TABLESPACE pg_default;

ALTER TABLE app.school_resources
    OWNER to postgres;
